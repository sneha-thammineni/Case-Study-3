"use strict";
 
var _interopRequire = function (obj) { return obj && obj.__esModule ? obj["default"] : obj; };
 
var expect = _interopRequire(require("expect"));
 
var TestUtils = _interopRequire(require("react-addons-test-utils"));
 
var expectJSX = _interopRequire(require("expect-jsx"));
 
var React = _interopRequire(require("react"));
 
var SearchBox = _interopRequire(require("../SearchBox"));
 
var sd = _interopRequire(require("skin-deep"));
 
expect.extend(expectJSX);
 
describe("SearchBox", function () {
	var tree = undefined,
	    instance = undefined,
	    vdom = undefined,
	    spy = undefined;
 
	function render(el) {
		tree = sd.shallowRender(el);
		instance = tree.getMountedInstance();
		vdom = tree.getRenderOutput();
	}
 
	beforeEach(function () {
		render(React.createElement(SearchBox, null));
		spy = expect.createSpy(function () {
			return null;
		});
	});
 
	it("renders a text input", function () {
		expect(vdom.type).toEqual("input");
		expect(vdom.props.type).toEqual("text");
	});
 
	it("renders the value", function () {
		var expected = "General Hux";
		render(React.createElement(SearchBox, { value: expected }));
		expect(vdom.props.value).toEqual(expected);
	});
 
	it("triggers the onSearch callback", function () {
		render(React.createElement(SearchBox, { onSearch: spy }));
		tree.props.onChange({ target: { value: "Star Wars" } });
		expect(spy).toHaveBeenCalledWith("Star Wars");
	});
 
	it("has a noop onSearch callback by default", function () {
		expect(instance.props.onSearch()).toEqual(null);
	});
 
	it("triggers the onClear callback", function () {
		render(React.createElement(SearchBox, { onClear: spy, value: "Star Wars" }));
		tree.props.onChange({ target: { value: "" } });
		expect(spy).toHaveBeenCalled();
	});
 
	it("has a noop onClear callback by default", function () {
		expect(instance.props.onClear()).toEqual(null);
	});
 
	it("affects the input's fontSize by windowWidth", function () {
		expect(tree.props.style.fontSize).toEqual("3rem");
		tree.reRender(React.createElement(SearchBox, { windowWidth: 300 }));
		expect(tree.props.style.fontSize).toEqual("1.5rem");
	});
 
	afterEach(function () {
		spy.restore();
		expect.restoreSpies();
	});
});
